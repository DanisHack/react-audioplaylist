## Install

```bash
npm install --save react-audioplaylist
```

## Usage

```jsx
import React, { Component } from 'react'

import AudioPlayer from 'react-audioplaylist'

class Example extends Component {
  render () {
    return (
      <AudioPlayer data={this.state.data} />
    )
  }
}
```
